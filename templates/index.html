# Re-import everything after reset to render the 2-minute Rain + Piano preview

from moviepy.editor import *

from pydub.generators import Sine, WhiteNoise

from pydub import AudioSegment


# Settings

duration = 120Â  # 2 minutes

fps = 24

resolution = (1280, 720)


# Load image

image_path = "/mnt/data/A_digital_photograph_captures_a_rain-soaked_window.png"

bg = ImageClip(image_path).set_duration(duration).resize(width=resolution[0]).set_fps(fps)


# Generate soft rain + ambient piano tone

rain = WhiteNoise().to_audio_segment(duration=duration * 1000).low_pass_filter(1200).apply_gain(-25)

piano = Sine(432).to_audio_segment(duration=duration * 1000).apply_gain(-28)

audio_mix = rain.overlay(piano)


# Export audio

audio_path = "/mnt/data/rain_piano_full_preview_audio.mp3"

audio_mix.export(audio_path, format="mp3")


# Load audio

audio_clip = AudioFileClip(audio_path)

video = bg.set_audio(audio_clip)


# Export preview video

preview_video_path = "/mnt/data/soothingvibes_rain_piano_full_preview.mp4"

video.write_videofile(preview_video_path, codec='libx264', audio_codec='aac', fps=fps)
